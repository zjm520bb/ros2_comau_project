ROUTINE request_abort
VAR
  release_path_wait : BOOLEAN
BEGIN
  release_path_wait := path_is_waiting
  motion_aborted := TRUE
  fly_queue_type := QUEUE_NONE
  fly_queue_count := 0
  path_upload_active := FALSE
  path_node_pending := FALSE
  path_state := PATH_ABORTED

  IF release_path_wait THEN
    IF active_path_type = PATH_CART THEN
      SIGNAL SEGMENT cart_path
    ELSE
      SIGNAL SEGMENT joint_path
    ENDIF
  ENDIF

  CANCEL ALL
  path_is_waiting := FALSE

  WRITE LUN_CRT ( '[MOV] Abort requested. Queues stopped', NL )
END request_abort


ROUTINE reset_abort
BEGIN
  motion_aborted := FALSE
END reset_abort


ROUTINE check_aborted : BOOLEAN
BEGIN
  IF motion_aborted THEN
    RETURN(TRUE)
  ENDIF

  RETURN(FALSE)
END check_aborted


ROUTINE str_2_real(str : STRING) : REAL
VAR
  len         : INTEGER
  i           : INTEGER
  cur         : INTEGER
  decimal_div : REAL
  real_num    : REAL
  negative    : BOOLEAN
  dot_found   : BOOLEAN
  digit_found : BOOLEAN
BEGIN
  parse_ok := FALSE
  real_num := 0
  decimal_div := 1
  negative := FALSE
  dot_found := FALSE
  digit_found := FALSE

  len := STR_LEN(str)

  IF len = 0 THEN
    RETURN(0)
  ENDIF

  i := 1

  IF ORD(str, 1) = 45 THEN
    negative := TRUE
    i := 2

    IF len = 1 THEN
      RETURN(0)
    ENDIF
  ENDIF

  WHILE i <= len DO
    cur := ORD(str, i)

    IF cur = 46 THEN
      IF dot_found THEN
        RETURN(0)
      ENDIF

      dot_found := TRUE
    ELSE
      IF (cur < 48) OR (cur > 57) THEN
        RETURN(0)
      ENDIF

      digit_found := TRUE

      IF NOT dot_found THEN
        real_num := real_num * 10 + cur - 48
      ELSE
        decimal_div := decimal_div * 10
        real_num := real_num + (cur - 48) / decimal_div
      ENDIF
    ENDIF

    i := i + 1
  ENDWHILE

  IF NOT digit_found THEN
    RETURN(0)
  ENDIF

  IF negative THEN
    real_num := -real_num
  ENDIF

  parse_ok := TRUE

  RETURN(real_num)
END str_2_real


ROUTINE get_command(command, action : STRING)
VAR
  delim : INTEGER
BEGIN
  action := ''
  delim := STR_LOC(command, ':')

  IF delim <> 0 THEN
    STR_XTRT( command, 1, delim - 1, action )
  ELSE
    action := command
  ENDIF
END get_command


ROUTINE get_parameter(command : STRING; param : ARRAY[*] OF REAL) : INTEGER
VAR
  delim : INTEGER
  x     : INTEGER
  coord : STRING[25]
  val   : REAL
BEGIN
  delim := STR_LOC(command, ':')

  IF delim = 0 THEN
    RETURN(0)
  ENDIF

  STR_DEL(command, 1, delim)

  IF STR_LEN(command) = 0 THEN
    RETURN(0)
  ENDIF

  FOR x := 1 TO paramlength DO
    coord := ''

    IF STR_LEN(command) = 0 THEN
      RETURN(x - 1)
    ENDIF

    delim := STR_LOC(command, ',')

    IF delim <> 0 THEN
      IF delim = 1 THEN
        WRITE LUN_CRT ( '[MOV] Empty parameter at index ', x, NL )

        RETURN(-1)
      ENDIF

      STR_XTRT( command, 1, delim - 1, coord )

      STR_DEL( command, 1, delim )
    ELSE
      STR_XTRT( command, 1, STR_LEN(command), coord )

      STR_DEL( command, 1, STR_LEN(command) )
    ENDIF

    val := str_2_real(coord)

    IF NOT parse_ok THEN
      WRITE LUN_CRT ( '[MOV] Parse error at parameter ', x, NL )

      WRITE LUN_TP ( '[MOV] Parameter parse error', NL )

      RETURN(-1)
    ENDIF

    param[x] := val
  ENDFOR

  IF STR_LEN(command) <> 0 THEN
    WRITE LUN_CRT ( '[MOV] Too many parameters', NL )

    RETURN(-2)
  ENDIF

  RETURN(paramlength)
END get_parameter


ROUTINE check_par_num(actual_num, required_num : INTEGER) : BOOLEAN
BEGIN
  IF actual_num = required_num THEN
    RETURN(TRUE)
  ENDIF

  WRITE LUN_CRT ( '[MOV] Wrong parameter number. Expected=', required_num, ' Received=', actual_num, NL )

  RETURN(FALSE)
END check_par_num

ROUTINE move_joint_to(p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER

  target : JOINTPOS
BEGIN
  IF NOT check_par_num(paramnumber, 6) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF check_aborted THEN
    RETURN(RESULT_ABORTED)

  target[2] := p2
  target[3] := p3

  target[5] := p5

    RETURN(RESULT_ABORTED)
  ENDIF

  RETURN(RESULT_OK)

ROUTINE move_relative_to(dx, dy, dz, frame_mode : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 4) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF check_aborted THEN
    RETURN(RESULT_ABORTED)
  ENDIF

  IF (dx = 0) AND (dy = 0) AND (dz = 0) THEN
    WRITE LUN_CRT ( '[MOV] RELATIVE vector must not be zero', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF frame_mode = FRAME_BASE THEN
    MOVE RELATIVE VEC(dx, dy, dz) IN BASE
  ELSE
    IF frame_mode = FRAME_TOOL THEN
      MOVE RELATIVE VEC(dx, dy, dz) IN TOOL
    ELSE
      IF frame_mode = FRAME_UFRAME THEN
        MOVE RELATIVE VEC(dx, dy, dz) IN UFRAME
      ELSE
        WRITE LUN_CRT ( '[MOV] Invalid RELATIVE frame. Use 0,1,2', NL )

        RETURN(RESULT_UNAVAILABLE)
      ENDIF
    ENDIF
  ENDIF

  IF check_aborted THEN
    RETURN(RESULT_ABORTED)
  ENDIF

  RETURN(RESULT_OK)
END move_relative_to

    IF frame_mode = FRAME_TOOL THEN
      MOVE ABOUT VEC(vx, vy, vz) BY angle IN TOOL
    ELSE

        MOVE ABOUT VEC(vx, vy, vz) BY angle IN UFRAME
      ELSE

        RETURN(RESULT_UNAVAILABLE)
      ENDIF

  ENDIF

  IF check_aborted THEN
    RETURN(RESULT_ABORTED)
  ENDIF

  RETURN(RESULT_OK)
END move_about_to

ROUTINE add_fly_linear(p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 6) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ( '[MOV] FLY queue is full', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_CART
  ENDIF

  IF fly_queue_type <> QUEUE_CART THEN
    WRITE LUN_CRT ( '[MOV] Cannot mix CART and JOINT queue', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  fly_queue_count := fly_queue_count + 1

  fly_segment_type[fly_queue_count] := SEG_LINEAR
  fly_target_queue[fly_queue_count] := POS(p1, p2, p3, p4, p5, p6)

  WRITE LUN_CRT ( '[MOV] LINEAR segment added. Count=', fly_queue_count, NL )

  RETURN(RESULT_OK)
END add_fly_linear


ROUTINE add_fly_circular(v1, v2, v3, v4, v5, v6 : REAL; p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 12) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ( '[MOV] FLY queue is full', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_CART
  ENDIF

  IF fly_queue_type <> QUEUE_CART THEN
    WRITE LUN_CRT ( '[MOV] Cannot mix CART and JOINT queue', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  fly_queue_count := fly_queue_count + 1

  fly_segment_type[fly_queue_count] := SEG_CIRCULAR

  fly_via_queue[fly_queue_count] := POS(v1, v2, v3, v4, v5, v6)

  fly_target_queue[fly_queue_count] := POS(p1, p2, p3, p4, p5, p6)

  WRITE LUN_CRT ( '[MOV] CIRCULAR segment added. Count=', fly_queue_count, NL )

  RETURN(RESULT_OK)
END add_fly_circular


ROUTINE add_fly_joint(p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
VAR
  target : JOINTPOS
BEGIN
  IF NOT check_par_num(paramnumber, 6) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ( '[MOV] FLY queue is full', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_JOINT
  ENDIF

  IF fly_queue_type <> QUEUE_JOINT THEN
    WRITE LUN_CRT ( '[MOV] Cannot mix JOINT and CART queue', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  target[1] := p1
  target[2] := p2
  target[3] := p3
  target[4] := p4
  target[5] := p5
  target[6] := p6

  fly_queue_count := fly_queue_count + 1
  fly_jnt_queue[fly_queue_count] := target

  WRITE LUN_CRT ( '[MOV] JOINT point added. Count=', fly_queue_count, NL )

  RETURN(RESULT_OK)
END add_fly_joint


ROUTINE execute_fly_queue : INTEGER
VAR
  i          : INTEGER
  last_index : INTEGER
BEGIN
  IF fly_queue_count < 2 THEN
    WRITE LUN_CRT ( '[MOV] At least two queue elements required', NL )

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF check_aborted THEN
    clear_fly_queue
    RETURN(RESULT_ABORTED)
  ENDIF

  WRITE LUN_CRT ( '[MOV] Executing FLY queue. Count=', fly_queue_count, NL )

  IF fly_queue_type = QUEUE_CART THEN
    IF $FLY_TYPE <> FLY_CART THEN
      WRITE LUN_CRT ( '[MOV] CART queue requires FLY_CART', NL )

      RETURN(RESULT_UNAVAILABLE)
    ENDIF

    FOR i := 1 TO fly_queue_count - 1 DO
      IF check_aborted THEN
        clear_fly_queue
        RETURN(RESULT_ABORTED)
      ENDIF

      IF fly_segment_type[i] = SEG_LINEAR THEN
        MOVEFLY LINEAR TO fly_target_queue[i] ADVANCE
      ELSE
        IF fly_segment_type[i] = SEG_CIRCULAR THEN
          MOVEFLY CIRCULAR TO fly_target_queue[i] VIA fly_via_queue[i] ADVANCE
        ELSE
          WRITE LUN_CRT ( '[MOV] Invalid CART segment type', NL )

          RETURN(RESULT_UNAVAILABLE)
        ENDIF
      ENDIF

      IF check_aborted THEN
        clear_fly_queue
        RETURN(RESULT_ABORTED)
      ENDIF
    ENDFOR

    last_index := fly_queue_count

    IF check_aborted THEN
      clear_fly_queue
      RETURN(RESULT_ABORTED)
    ENDIF

    IF fly_segment_type[last_index] = SEG_LINEAR THEN
      MOVE LINEAR TO fly_target_queue[last_index]
    ELSE
      IF fly_segment_type[last_index] = SEG_CIRCULAR THEN
        MOVE CIRCULAR TO fly_target_queue[last_index] VIA fly_via_queue[last_index]
      ELSE
        WRITE LUN_CRT ( '[MOV] Invalid final CART segment type', NL )

        RETURN(RESULT_UNAVAILABLE)
      ENDIF
    ENDIF

    IF check_aborted THEN
      clear_fly_queue
      RETURN(RESULT_ABORTED)
    ENDIF

    clear_fly_queue

    RETURN(RESULT_OK)
  ENDIF

  IF fly_queue_type = QUEUE_JOINT THEN
    IF $FLY_TYPE = FLY_CART THEN
      WRITE LUN_CRT ( '[MOV] JOINT queue requires FLY_NORM', NL )

      RETURN(RESULT_UNAVAILABLE)
    ENDIF

    FOR i := 1 TO fly_queue_count - 1 DO
      IF check_aborted THEN
        clear_fly_queue
        RETURN(RESULT_ABORTED)
      ENDIF

      MOVEFLY JOINT TO fly_jnt_queue[i] ADVANCE

      IF check_aborted THEN
        clear_fly_queue
        RETURN(RESULT_ABORTED)
      ENDIF
    ENDFOR

    IF check_aborted THEN
      clear_fly_queue
      RETURN(RESULT_ABORTED)
    ENDIF

    MOVE JOINT TO fly_jnt_queue[fly_queue_count]

    IF check_aborted THEN
      clear_fly_queue
      RETURN(RESULT_ABORTED)
    ENDIF

    clear_fly_queue

    RETURN(RESULT_OK)
  ENDIF

  WRITE LUN_CRT ( '[MOV] Invalid queue type', NL )

  RETURN(RESULT_UNAVAILABLE)
END execute_fly_queue


ROUTINE clear_native_paths
VAR
  count : INTEGER
  idx   : INTEGER
BEGIN
  count := PATH_LEN(cart_path)
  IF count > 0 THEN
    NODE_DEL(cart_path, 1, count)
  ENDIF

  count := PATH_LEN(joint_path)
  IF count > 0 THEN
    NODE_DEL(joint_path, 1, count)
  ENDIF

  FOR idx := 1 TO 32 DO
    cart_path.COND_TBL[idx] := 0
    joint_path.COND_TBL[idx] := 0
  ENDFOR
END clear_native_paths

ROUTINE begin_path(path_id, path_type, node_count, first_node, last_node : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 5) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (path_id <= 0) OR (node_count <= 0) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (path_type <> PATH_CART) AND (path_type <> PATH_JOINT) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (first_node < 1) OR (last_node < 1) OR (first_node > node_count) OR (last_node > node_count) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  clear_native_paths

  active_path_id := ROUND(path_id)
  active_path_type := ROUND(path_type)
  expected_path_nodes := ROUND(node_count)
  committed_path_nodes := 0
  pending_path_node := 0
  path_start_index := ROUND(first_node)
  path_end_index := ROUND(last_node)
  path_current_node := 0
  path_upload_active := TRUE
  path_node_pending := FALSE
  path_is_waiting := FALSE
  path_state := PATH_UPLOADING

  RETURN(RESULT_OK)
END begin_path


ROUTINE set_path_frame(path_id, frame_index, p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 8) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (NOT path_upload_active) OR (ROUND(path_id) <> active_path_id) OR (active_path_type <> PATH_CART) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(frame_index)
  IF (idx < 1) OR (idx > 7) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  cart_path.FRM_TBL[idx] := POS( p1, p2, p3, p4, p5, p6 )
  RETURN(RESULT_OK)
END set_path_frame


ROUTINE set_path_condition(path_id, slot, handler : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 3) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (NOT path_upload_active) OR (ROUND(path_id) <> active_path_id) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(slot)
  IF (idx < 1) OR (idx > 32) OR (handler <= 0) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    cart_path.COND_TBL[idx] := ROUND(handler)
  ELSE
    joint_path.COND_TBL[idx] := ROUND(handler)
  ENDIF
  RETURN(RESULT_OK)
END set_path_condition


ROUTINE path_node_target(path_id, node_index, motion_type, p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx         : INTEGER
  target      : JOINTPOS
  target_pos  : POSITION
  current_pos : POSITION
  cnfg        : STRING[33]
BEGIN
  IF NOT check_par_num(paramnumber, 9) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(node_index)
  IF (NOT path_upload_active) OR path_node_pending OR (ROUND(path_id) <> active_path_id) OR (idx <> committed_path_nodes + 1) OR (idx > expected_path_nodes) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    NODE_APP(cart_path)
    target_pos := POS(p1, p2, p3, p4, p5, p6)
    current_pos := ARM_POS(1)
    cnfg := POS_GET_CNFG(current_pos)
    POS_SET_CNFG(target_pos, cnfg)
    cart_path.NODE[idx].$MAIN_POS := target_pos
    cart_path.NODE[idx].$CNFG_CARE := FALSE
    cart_path.NODE[idx].$ORNT_TYPE := WRIST_JNT
    cart_path.NODE[idx].$SPD_OPT := SPD_LIN
    cart_path.NODE[idx].$SING_CARE := FALSE
    cart_path.NODE[idx].$TURN_CARE := FALSE

    IF motion_type = 0 THEN
      cart_path.NODE[idx].$MOVE_TYPE := JOINT
    ELSE
      IF motion_type = 1 THEN
        cart_path.NODE[idx].$MOVE_TYPE := LINEAR
      ELSE
        IF motion_type = 2 THEN
          cart_path.NODE[idx].$MOVE_TYPE := CIRCULAR
        ELSE
          IF motion_type = 3 THEN
            cart_path.NODE[idx].$MOVE_TYPE := SEG_VIA
          ELSE
            RETURN(RESULT_UNAVAILABLE)
          ENDIF
        ENDIF
      ENDIF
    ENDIF
  ELSE
    IF motion_type <> 0 THEN
      RETURN(RESULT_UNAVAILABLE)
    ENDIF

    target[1] := p1
    target[2] := p2
    target[3] := p3
    target[4] := p4
    target[5] := p5
    target[6] := p6
    NODE_APP(joint_path)
    joint_path.NODE[idx].$MAIN_JNTP := target
    joint_path.NODE[idx].$MOVE_TYPE := JOINT
  ENDIF

  pending_path_node := idx
  path_node_pending := TRUE
  path_target_set := TRUE
  path_motion_set := FALSE
  path_blend_set := FALSE
  path_sync_set := FALSE
  RETURN(RESULT_OK)
END path_node_target


ROUTINE path_node_motion(path_id, node_index, lin_speed, rot_speed, seg_ovr, term_type, tolerance, seg_data : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 8) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(node_index)
  IF (NOT path_node_pending) OR (ROUND(path_id) <> active_path_id) OR (idx <> pending_path_node) OR (seg_ovr <= 0) OR (seg_ovr > 100) OR (term_type < 0) OR (term_type > 4) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    cart_path.NODE[idx].$LIN_SPD := lin_speed
    cart_path.NODE[idx].$ROT_SPD := rot_speed
    cart_path.NODE[idx].$SEG_OVR := ROUND(seg_ovr)
    cart_path.NODE[idx].$SEG_TOL := tolerance

    IF term_type = 0 THEN
      cart_path.NODE[idx].$SEG_TERM_TYPE := COARSE
    ELSE
      IF term_type = 1 THEN
        cart_path.NODE[idx].$SEG_TERM_TYPE := FINE
      ELSE
        IF term_type = 2 THEN
          cart_path.NODE[idx].$SEG_TERM_TYPE := JNT_COARSE
        ELSE
          IF term_type = 3 THEN
            cart_path.NODE[idx].$SEG_TERM_TYPE := JNT_FINE
          ELSE
            cart_path.NODE[idx].$SEG_TERM_TYPE := NOSETTLE
          ENDIF
        ENDIF
      ENDIF
    ENDIF

    cart_path.NODE[idx].$SEG_DATA := (seg_data <> 0)
  ELSE
    joint_path.NODE[idx].$SEG_OVR := ROUND(seg_ovr)
    joint_path.NODE[idx].$SEG_TOL := tolerance

    IF term_type = 0 THEN
      joint_path.NODE[idx].$SEG_TERM_TYPE := COARSE
    ELSE
      IF term_type = 1 THEN
        joint_path.NODE[idx].$SEG_TERM_TYPE := FINE
      ELSE
        IF term_type = 2 THEN
          joint_path.NODE[idx].$SEG_TERM_TYPE := JNT_COARSE
        ELSE
          IF term_type = 3 THEN
            joint_path.NODE[idx].$SEG_TERM_TYPE := JNT_FINE
          ELSE
            joint_path.NODE[idx].$SEG_TERM_TYPE := NOSETTLE
          ENDIF
        ENDIF
      ENDIF
    ENDIF

    joint_path.NODE[idx].$SEG_DATA := (seg_data <> 0)
  ENDIF

  path_motion_set := TRUE
  RETURN(RESULT_OK)
END path_node_motion


ROUTINE path_node_blend(path_id, node_index, use_fly, fly_type, fly_per, fly_dist, fly_traj, stress_per : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 8) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(node_index)
  IF (NOT path_node_pending) OR (ROUND(path_id) <> active_path_id) OR (idx <> pending_path_node) OR (fly_type < 0) OR (fly_type > 1) OR (fly_traj < 0) OR (fly_traj > 3) OR (fly_dist < 0) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    cart_path.NODE[idx].$SEG_FLY := (use_fly <> 0)
    cart_path.NODE[idx].$SEG_FLY_PER := ROUND(fly_per)
    cart_path.NODE[idx].$SEG_FLY_DIST := fly_dist
    cart_path.NODE[idx].$SEG_STRESS_PER := ROUND(stress_per)

    IF fly_type = 0 THEN
      cart_path.NODE[idx].$SEG_FLY_TYPE := FLY_NORM
    ELSE
      cart_path.NODE[idx].$SEG_FLY_TYPE := FLY_CART
    ENDIF

    IF fly_traj = 0 THEN
      cart_path.NODE[idx].$SEG_FLY_TRAJ := FLY_AUTO
    ELSE
      IF fly_traj = 1 THEN
        cart_path.NODE[idx].$SEG_FLY_TRAJ := FLY_TOL
      ELSE
        IF fly_traj = 2 THEN
          cart_path.NODE[idx].$SEG_FLY_TRAJ := FLY_PASS
        ELSE
          cart_path.NODE[idx].$SEG_FLY_TRAJ := FLY_FROM
        ENDIF
      ENDIF
    ENDIF
  ELSE
    joint_path.NODE[idx].$SEG_FLY := (use_fly <> 0)
    joint_path.NODE[idx].$SEG_FLY_PER := ROUND(fly_per)
    IF fly_type = 0 THEN
      joint_path.NODE[idx].$SEG_FLY_TYPE := FLY_NORM
    ELSE
      joint_path.NODE[idx].$SEG_FLY_TYPE := FLY_CART
    ENDIF
  ENDIF

  path_blend_set := TRUE
  RETURN(RESULT_OK)
END path_node_blend


ROUTINE path_node_sync(path_id, node_index, ref_idx, tool_idx, cond_mask, cond_mask_back, seg_wait : REAL; paramnumber : INTEGER) : INTEGER
VAR
  idx : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 7) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  idx := ROUND(node_index)
  IF (NOT path_node_pending) OR (ROUND(path_id) <> active_path_id) OR (idx <> pending_path_node) OR (ref_idx < 0) OR (ref_idx > 7) OR (tool_idx < 0) OR (tool_idx > 7) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    cart_path.NODE[idx].$SEG_REF_IDX := ROUND(ref_idx)
    cart_path.NODE[idx].$SEG_TOOL_IDX := ROUND(tool_idx)
    cart_path.NODE[idx].$COND_MASK := ROUND(cond_mask)
    cart_path.NODE[idx].$COND_MASK_BACK := ROUND(cond_mask_back)
    cart_path.NODE[idx].$SEG_WAIT := (seg_wait <> 0)
  ELSE
    IF (ref_idx <> 0) OR (tool_idx <> 0) THEN
      RETURN(RESULT_UNAVAILABLE)
    ENDIF
    joint_path.NODE[idx].$COND_MASK := ROUND(cond_mask)
    joint_path.NODE[idx].$COND_MASK_BACK := ROUND(cond_mask_back)
    joint_path.NODE[idx].$SEG_WAIT := (seg_wait <> 0)
  ENDIF

  path_sync_set := TRUE
  RETURN(RESULT_OK)
END path_node_sync


ROUTINE commit_path_node(path_id, node_index : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 2) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (NOT path_node_pending) OR (ROUND(path_id) <> active_path_id) OR (ROUND(node_index) <> pending_path_node) OR (NOT path_target_set) OR (NOT path_motion_set) OR (NOT path_blend_set) OR (NOT path_sync_set) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  committed_path_nodes := committed_path_nodes + 1
  path_node_pending := FALSE
  pending_path_node := 0
  RETURN(RESULT_OK)
END commit_path_node


ROUTINE commit_path(path_id, node_count : REAL; paramnumber : INTEGER) : INTEGER
VAR
  actual_count : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 2) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (NOT path_upload_active) OR path_node_pending OR (ROUND(path_id) <> active_path_id) OR (ROUND(node_count) <> expected_path_nodes) OR (committed_path_nodes <> expected_path_nodes) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF active_path_type = PATH_CART THEN
    actual_count := PATH_LEN(cart_path)
  ELSE
    actual_count := PATH_LEN(joint_path)
  ENDIF

  IF actual_count <> expected_path_nodes THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  path_upload_active := FALSE
  path_state := PATH_READY
  RETURN(RESULT_OK)
END commit_path


ROUTINE abort_path_upload(path_id : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 1) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF ROUND(path_id) <> active_path_id THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  clear_native_paths
  path_upload_active := FALSE
  path_node_pending := FALSE
  path_is_waiting := FALSE
  expected_path_nodes := 0
  committed_path_nodes := 0
  path_current_node := 0
  path_state := PATH_ABORTED
  RETURN(RESULT_OK)
END abort_path_upload


ROUTINE signal_path_segment(path_id, expected_node : INTEGER) : INTEGER
BEGIN
  IF (path_id <> active_path_id) OR (NOT path_is_waiting) THEN
    RETURN(0)
  ENDIF

  IF (expected_node <> 0) AND (expected_node <> path_current_node) THEN
    RETURN(0)
  ENDIF

  IF active_path_type = PATH_CART THEN
    SIGNAL SEGMENT cart_path
  ELSE
    SIGNAL SEGMENT joint_path
  ENDIF

  path_is_waiting := FALSE
  path_state := PATH_EXECUTING
  RETURN(1)
END signal_path_segment


ROUTINE execute_path(path_id, first_node, last_node : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 3) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (path_state <> PATH_READY) OR (ROUND(path_id) <> active_path_id) OR (ROUND(first_node) <> path_start_index) OR (ROUND(last_node) <> path_end_index) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF check_aborted THEN
    RETURN(RESULT_ABORTED)
  ENDIF

  path_state := PATH_EXECUTING
  path_is_waiting := FALSE
  path_current_node := path_start_index

  IF active_path_type = PATH_CART THEN
    MOVE ALONG cart_path[path_start_index..path_end_index]
  ELSE
    MOVE ALONG joint_path[path_start_index..path_end_index]
  ENDIF

  IF check_aborted THEN
    path_state := PATH_ABORTED
    RETURN(RESULT_ABORTED)
  ENDIF

  path_current_node := path_end_index
  path_state := PATH_DONE
  RETURN(RESULT_OK)
END execute_path

    WRITE path_feedback_lun ( 'PATH_WAITING:', active_path_id, ',', path_current_node, NL )
  ENDIF
END report_path_wait

ROUTINE call_service(command : STRING) : INTEGER
VAR
  param       : ARRAY[paramlength] OF REAL
  action      : STRING[25]
  paramnumber : INTEGER
BEGIN
  action := ''
  paramnumber := 0

  get_command(command, action)

  IF action = 'Hello' THEN
    RETURN(hello)
  ENDIF

  IF action = 'clearFlyQueue' THEN
    RETURN(clear_fly_queue)
  ENDIF

  IF action = 'executeFlyQueue' THEN
    RETURN(execute_fly_queue)
  ENDIF

  paramnumber := get_parameter(command, param)

  IF paramnumber < 0 THEN
    RETURN(RESULT_PARSE_ERROR)
  ENDIF

  IF action = 'beginPath' THEN
    RETURN( begin_path( param[1], param[2], param[3], param[4], param[5], paramnumber ) )
  ENDIF

  IF action = 'setPathFrame' THEN
    RETURN( set_path_frame( param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], paramnumber ) )
  ENDIF

  IF action = 'setPathCondition' THEN
    RETURN( set_path_condition( param[1], param[2], param[3], paramnumber ) )
  ENDIF

    RETURN( path_node_target( param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], param[9], paramnumber ) )
  ENDIF

  IF action = 'pathNodeMotion' THEN
    RETURN( path_node_motion( param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], paramnumber ) )
  ENDIF

  IF action = 'pathNodeBlend' THEN
    RETURN( path_node_blend( param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], paramnumber ) )
  ENDIF

  IF action = 'pathNodeSync' THEN
    RETURN( path_node_sync( param[1], param[2], param[3], param[4], param[5], param[6], param[7], paramnumber ) )
  ENDIF

  IF action = 'commitPathNode' THEN
    RETURN( commit_path_node( param[1], param[2], paramnumber ) )
  ENDIF

  IF action = 'commitPath' THEN
    RETURN( commit_path( param[1], param[2], paramnumber ) )
  ENDIF

  IF action = 'abortPathUpload' THEN
    RETURN( abort_path_upload( param[1], paramnumber ) )
  ENDIF

  IF action = 'executePath' THEN
    RETURN( execute_path( param[1], param[2], param[3], paramnumber ) )
  ENDIF

  IF action = 'setBase' THEN
    RETURN( set_base_to( param[1], param[2], param[3], param[4], param[5], param[6], paramnumber ) )
  ENDIF

  IF (action = 'setTool') OR (action = 'tool') THEN

    RETURN( set_tool_to( param[1], param[2], param[3], param[4], param[5], param[6], paramnumber ) )
  ENDIF

  IF (action = 'setUframe') OR (action = 'userFrame') THEN

    RETURN( set_uframe_to( param[1], param[2], param[3], param[4], param[5], param[6], paramnumber ) )
  ENDIF

  IF action = 'setSpeedLin' THEN
    RETURN( set_speed_linear_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setSpeedJnt' THEN
    RETURN( set_speed_joint_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setJointOverrides' THEN
    RETURN( set_joint_overrides( param[1], param[2], param[3], param[4], param[5], param[6], paramnumber ) )
  ENDIF

  IF action = 'setAcceleration' THEN
    RETURN( set_acceleration_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setDeceleration' THEN
    RETURN( set_deceleration_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setOrientation' THEN
    RETURN( set_orientation_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setTermination' THEN
    RETURN( set_termination_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setFlyNorm' THEN
    RETURN( set_fly_norm_to( param[1], paramnumber ) )
  ENDIF

  IF action = 'setFlyCart' THEN
    RETURN( set_fly_cart_to( param[1], param[2], param[3], paramnumber ) )
  ENDIF
