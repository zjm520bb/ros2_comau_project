PROGRAM PDL2_ROS2_main &PA, STACK = 5000

CONST
  server_port   = 8000
  buffer_length = 256

  SETUP_STATE             = 0
  IDLE_STATE              = 10
  PERMISSION_STATE        = 20
  MOVEMENT_STATE          = 30
  ERROR_STATE             = 90

VAR
  state   : INTEGER
  command : STRING[buffer_length]
  BR_PC   : INTEGER


ROUTINE accept_client(PORT : INTEGER) : INTEGER EXPORTED FROM CONNECT_VIA_TCP_C4G
ROUTINE disconnect_from(LUN : INTEGER) : INTEGER EXPORTED FROM CONNECT_VIA_TCP_C4G
ROUTINE send_to_LUN(LUN : INTEGER; msg : STRING) : INTEGER EXPORTED FROM CONNECT_VIA_TCP_C4G
ROUTINE read_from_LUN(LUN : INTEGER) : STRING EXPORTED FROM CONNECT_VIA_TCP_C4G

ROUTINE call_service(command : STRING) : INTEGER EXPORTED FROM ROS2_MOVEMENTS


ROUTINE display(printable : STRING) : INTEGER
BEGIN
  WRITE LUN_CRT (printable, NL)
  WRITE LUN_TP  (printable, NL)
  RETURN(0)
END display


ROUTINE safe_disconnect : INTEGER
BEGIN
  IF BR_PC <> 0 THEN
    disconnect_from(BR_PC)
    BR_PC := 0
  ENDIF
  RETURN(0)
END safe_disconnect


ROUTINE setup : BOOLEAN
BEGIN
  safe_disconnect
  BR_PC := accept_client(server_port)

  IF BR_PC = -1 THEN
    RETURN(FALSE)
  ELSE
    display('[MAIN] Client connected on port 8000')
    RETURN(TRUE)
  ENDIF
END setup


ROUTINE get_command : STRING
BEGIN
  RETURN(read_from_LUN(BR_PC))
END get_command


-- Two-step handshake:
-- 1) echo command back
-- 2) PC must reply 'start' to proceed
ROUTINE can_start(cmd : STRING) : INTEGER
VAR
  ans : STRING[buffer_length]
  flush : STRING[buffer_length]
BEGIN
  send_to_LUN(BR_PC, cmd)
  DELAY 50
  ans := read_from_LUN(BR_PC)

  IF ans = '' THEN
    ans := read_from_LUN(BR_PC)
  ENDIF

  IF ans = 'start' THEN
    RETURN(1)
  ENDIF

  RETURN(0)
END can_start


ROUTINE send_error
BEGIN
  send_to_LUN(BR_PC, 'ERROR')
END send_error

ROUTINE send_unavailable
BEGIN
  send_to_LUN(BR_PC, 'Movement unavailable')
END send_unavailable

ROUTINE send_finish
BEGIN
  send_to_LUN(BR_PC, 'Movement finished')
END send_finish


BEGIN
  BR_PC := 0
  command := ''

  display('[MAIN] ROS2 server started (port 8000)')
  state := SETUP_STATE

  WHILE TRUE DO
    DELAY 200

    SELECT state OF

      CASE (SETUP_STATE):
        IF setup THEN
          state := IDLE_STATE
        ELSE
          display('[MAIN] Connection failed (8000). HOLD.')
          HOLD
          state := SETUP_STATE
        ENDIF


      CASE (IDLE_STATE):
        command := ''
        command := get_command

        IF command = '' THEN
          state := IDLE_STATE
        ELSE
          display('[MAIN] Command received:')
          display(command)
          state := PERMISSION_STATE
        ENDIF


      CASE (PERMISSION_STATE):
        SELECT can_start(command) OF
          CASE (1):
            display('[MAIN] Starting command:')
            display(command)
            state := MOVEMENT_STATE
          ELSE:
            display('[MAIN] Start declined. Back to IDLE.')
            state := IDLE_STATE
        ENDSELECT


      CASE (MOVEMENT_STATE):
        -- Execute motion (may block on MOVE)
        SELECT call_service(command) OF
          CASE (1):
            display('[MAIN] Movement done!')
            send_finish
            state := IDLE_STATE

          CASE (0):
            display('[MAIN] Movement unavailable!')
            send_unavailable
            state := IDLE_STATE

          ELSE:
            display('[MAIN] Error during movement!')
            send_error
            state := ERROR_STATE
        ENDSELECT


      CASE (ERROR_STATE):
        display('[MAIN] Error state. HOLD.')
        HOLD
        state := IDLE_STATE


      ELSE:
        display('[MAIN] Unknown state. Reset.')
        safe_disconnect
        state := SETUP_STATE

    ENDSELECT

  ENDWHILE
END PDL2_ROS2_main
