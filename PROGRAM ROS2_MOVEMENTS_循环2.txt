PROGRAM ROS2_MOVEMENTS

CONST

  RESULT_OK             = 1
  RESULT_UNAVAILABLE    = 0
  RESULT_UNKNOWN_ACTION = -1
  RESULT_PARSE_ERROR    = -2

  QUEUE_NONE  = 0
  QUEUE_CART  = 1
  QUEUE_JOINT = 2

  SEG_NONE     = 0
  SEG_LINEAR   = 1
  SEG_CIRCULAR = 2

VAR
  parse_ok : BOOLEAN

  fly_queue_type  : INTEGER
  fly_queue_count : INTEGER

  fly_segment_type : ARRAY[max_fly_points] OF INTEGER
  fly_target_queue : ARRAY[max_fly_points] OF POSITION

  fly_jnt_queue    : ARRAY[max_fly_points] OF JOINTPOS

ROUTINE str_2_real(str : STRING) : REAL
VAR
  len         : INTEGER
  i           : INTEGER
  cur         : INTEGER
  decimal_div : REAL
  real_num    : REAL
  negative    : BOOLEAN
  dot_found   : BOOLEAN
  digit_found : BOOLEAN
BEGIN
  parse_ok := FALSE
  real_num := 0
  decimal_div := 1
  negative := FALSE
  dot_found := FALSE
  digit_found := FALSE

  len := STR_LEN(str)

  IF len = 0 THEN
    RETURN(0)
  ENDIF

  i := 1

  IF ORD(str, 1) = 45 THEN
    negative := TRUE
    i := 2

    IF len = 1 THEN
      RETURN(0)
    ENDIF
  ENDIF

  WHILE i <= len DO
    cur := ORD(str, i)

    IF cur = 46 THEN
      IF dot_found THEN
        RETURN(0)
      ENDIF

      dot_found := TRUE
    ELSE
      IF (cur < 48) OR (cur > 57) THEN
        RETURN(0)
      ENDIF

      digit_found := TRUE

      IF NOT dot_found THEN
        real_num := real_num * 10 + cur - 48
      ELSE
        decimal_div := decimal_div * 10
        real_num := real_num + (cur - 48) / decimal_div
      ENDIF
    ENDIF

    i := i + 1
  ENDWHILE

  IF NOT digit_found THEN
    RETURN(0)
  ENDIF

  IF negative THEN
    real_num := -real_num
  ENDIF

  parse_ok := TRUE
  RETURN(real_num)
END str_2_real


ROUTINE get_command(command, action : STRING)
VAR
  delim : INTEGER
BEGIN
  action := ''
  delim := STR_LOC(command, ':')

  IF delim <> 0 THEN
    STR_XTRT(command, 1, delim - 1, action)
  ELSE
    action := command
  ENDIF
END get_command


ROUTINE get_parameter(command : STRING; param : ARRAY[*] OF REAL) : INTEGER
VAR
  delim : INTEGER
  x     : INTEGER
  coord : STRING[25]
  val   : REAL
BEGIN
  delim := STR_LOC(command, ':')

  IF delim = 0 THEN
    RETURN(0)
  ENDIF

  STR_DEL(command, 1, delim)

  IF STR_LEN(command) = 0 THEN
    RETURN(0)
  ENDIF

  FOR x := 1 TO paramlength DO
    coord := ''

    IF STR_LEN(command) = 0 THEN
      RETURN(x - 1)
    ENDIF

    delim := STR_LOC(command, ',')

    IF delim <> 0 THEN
      IF delim = 1 THEN
        WRITE LUN_CRT ('[MOV] Empty parameter at index ', x, NL)
        RETURN(-1)
      ENDIF

      STR_XTRT(command, 1, delim - 1, coord)
      STR_DEL(command, 1, delim)
    ELSE
      STR_XTRT(command, 1, STR_LEN(command), coord)
      STR_DEL(command, 1, STR_LEN(command))
    ENDIF

    val := str_2_real(coord)

    IF NOT parse_ok THEN
      WRITE LUN_CRT ('[MOV] Parse error at parameter ', x, NL)
      WRITE LUN_TP ('[MOV] Parameter parse error', NL)
      RETURN(-1)
    ENDIF

    param[x] := val
  ENDFOR

  IF STR_LEN(command) <> 0 THEN
    WRITE LUN_CRT ('[MOV] Too many parameters', NL)
    RETURN(-2)
  ENDIF

  RETURN(paramlength)
END get_parameter


ROUTINE check_par_num(actual_num, required_num : INTEGER) : BOOLEAN
BEGIN
  IF actual_num = required_num THEN
    RETURN(TRUE)
  ENDIF

  WRITE LUN_CRT ('[MOV] Wrong parameter number. Expected=', required_num, ' Received=', actual_num, NL)
  RETURN(FALSE)
END check_par_num

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  $BASE := POS(p1, p2, p3, p4, p5, p6)
  WRITE LUN_CRT ('[MOV] BASE updated', NL)
  RETURN(RESULT_OK)

ROUTINE set_speed_joint_to(speed_per : REAL; paramnumber : INTEGER) : INTEGER
VAR
  speed_value : INTEGER
  i           : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 1) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (speed_per < 1) OR (speed_per > 100) THEN
    WRITE LUN_CRT ('[MOV] Joint override must be 1..100', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  speed_value := ROUND(speed_per)

  FOR i := 1 TO 6 DO
    $ARM_DATA[1].JNT_OVR[i] := speed_value
  ENDFOR

  WRITE LUN_CRT ('[MOV] Joint override set to ', speed_value, NL)
  RETURN(RESULT_OK)
END set_speed_joint_to

ROUTINE set_fly_cart_to(stress_per, traj_mode, fly_dist : REAL; paramnumber : INTEGER) : INTEGER

    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF (stress_per < 1) OR (stress_per > 100) THEN
    WRITE LUN_CRT ('[MOV] STRESS_PER must be 1..100', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_dist < 0 THEN
    WRITE LUN_CRT ('[MOV] FLY_DIST must be greater than or equal to zero', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  $FLY_TYPE := FLY_CART
  $SPD_OPT := SPD_LIN
  $STRESS_PER := ROUND(stress_per)
  $FLY_DIST := fly_dist

  IF traj_mode = 0 THEN
    $FLY_TRAJ := FLY_AUTO
  ELSE
    IF traj_mode = 1 THEN
      $FLY_TRAJ := FLY_TOL
    ELSE
      IF traj_mode = 2 THEN
        $FLY_TRAJ := FLY_PASS
      ELSE
        IF traj_mode = 3 THEN
          $FLY_TRAJ := FLY_FROM
        ELSE
          WRITE LUN_CRT ('[MOV] Invalid FLY_TRAJ mode', NL)
          RETURN(RESULT_UNAVAILABLE)
        ENDIF
      ENDIF
    ENDIF
  ENDIF

  WRITE LUN_CRT ('[MOV] FLY_CART configured', NL)
  RETURN(RESULT_OK)
END set_fly_cart_to

ROUTINE clear_fly_queue : INTEGER
BEGIN
  fly_queue_type := QUEUE_NONE
  fly_queue_count := 0

  WRITE LUN_CRT ('[MOV] FLY queue cleared', NL)

  RETURN(RESULT_OK)
END clear_fly_queue


ROUTINE add_fly_linear(p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 6) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ('[MOV] FLY queue is full', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_CART
  ENDIF

  IF fly_queue_type <> QUEUE_CART THEN
    WRITE LUN_CRT ('[MOV] Cannot mix CART and JOINT queue', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  fly_queue_count := fly_queue_count + 1

  fly_segment_type[fly_queue_count] := SEG_LINEAR
  fly_target_queue[fly_queue_count] := POS(p1, p2, p3, p4, p5, p6)

  WRITE LUN_CRT ('[MOV] LINEAR segment added. Count=', fly_queue_count, NL)

  RETURN(RESULT_OK)
END add_fly_linear


ROUTINE add_fly_circular(v1, v2, v3, v4, v5, v6, p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
BEGIN
  IF NOT check_par_num(paramnumber, 12) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ('[MOV] FLY queue is full', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_CART
  ENDIF

  IF fly_queue_type <> QUEUE_CART THEN
    WRITE LUN_CRT ('[MOV] Cannot mix CART and JOINT queue', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  fly_queue_count := fly_queue_count + 1

  fly_segment_type[fly_queue_count] := SEG_CIRCULAR
  fly_via_queue[fly_queue_count] := POS(v1, v2, v3, v4, v5, v6)
  fly_target_queue[fly_queue_count] := POS(p1, p2, p3, p4, p5, p6)

  WRITE LUN_CRT ('[MOV] CIRCULAR segment added. Count=', fly_queue_count, NL)

  RETURN(RESULT_OK)
END add_fly_circular


ROUTINE add_fly_joint(p1, p2, p3, p4, p5, p6 : REAL; paramnumber : INTEGER) : INTEGER
VAR
  target : JOINTPOS
BEGIN
  IF NOT check_par_num(paramnumber, 6) THEN
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_count >= max_fly_points THEN
    WRITE LUN_CRT ('[MOV] FLY queue is full', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  IF fly_queue_type = QUEUE_NONE THEN
    fly_queue_type := QUEUE_JOINT
  ENDIF

  IF fly_queue_type <> QUEUE_JOINT THEN
    WRITE LUN_CRT ('[MOV] Cannot mix JOINT and CART queue', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  target[1] := p1
  target[2] := p2
  target[3] := p3
  target[4] := p4
  target[5] := p5
  target[6] := p6

  fly_queue_count := fly_queue_count + 1
  fly_jnt_queue[fly_queue_count] := target

  WRITE LUN_CRT ('[MOV] JOINT point added. Count=', fly_queue_count, NL)

  RETURN(RESULT_OK)
END add_fly_joint


ROUTINE execute_fly_queue : INTEGER
VAR
  i          : INTEGER
  last_index : INTEGER
BEGIN
  IF fly_queue_count < 2 THEN
    WRITE LUN_CRT ('[MOV] At least two queue elements required', NL)
    RETURN(RESULT_UNAVAILABLE)
  ENDIF

  WRITE LUN_CRT ('[MOV] Executing FLY queue. Count=', fly_queue_count, NL)

  IF fly_queue_type = QUEUE_CART THEN
    IF $FLY_TYPE <> FLY_CART THEN
      WRITE LUN_CRT ('[MOV] CART queue requires FLY_CART', NL)
      RETURN(RESULT_UNAVAILABLE)
    ENDIF

    FOR i := 1 TO fly_queue_count - 1 DO
      IF fly_segment_type[i] = SEG_LINEAR THEN
        MOVEFLY LINEAR TO fly_target_queue[i] ADVANCE
      ELSE
        IF fly_segment_type[i] = SEG_CIRCULAR THEN
          MOVEFLY CIRCULAR TO fly_target_queue[i] VIA fly_via_queue[i] ADVANCE
        ELSE
          WRITE LUN_CRT ('[MOV] Invalid CART segment type', NL)
          RETURN(RESULT_UNAVAILABLE)
        ENDIF
      ENDIF
    ENDFOR

    last_index := fly_queue_count

    IF fly_segment_type[last_index] = SEG_LINEAR THEN
      MOVE LINEAR TO fly_target_queue[last_index]
    ELSE
      IF fly_segment_type[last_index] = SEG_CIRCULAR THEN
        MOVE CIRCULAR TO fly_target_queue[last_index] VIA fly_via_queue[last_index]
      ELSE
        WRITE LUN_CRT ('[MOV] Invalid final CART segment type', NL)
        RETURN(RESULT_UNAVAILABLE)
      ENDIF
    ENDIF

    clear_fly_queue

    RETURN(RESULT_OK)
  ENDIF

  IF fly_queue_type = QUEUE_JOINT THEN
    IF $FLY_TYPE = FLY_CART THEN
      WRITE LUN_CRT ('[MOV] JOINT queue requires FLY_NORM', NL)
      RETURN(RESULT_UNAVAILABLE)
    ENDIF

    FOR i := 1 TO fly_queue_count - 1 DO
      MOVEFLY JOINT TO fly_jnt_queue[i] ADVANCE
    ENDFOR

    MOVE JOINT TO fly_jnt_queue[fly_queue_count]

    clear_fly_queue

    RETURN(RESULT_OK)
  ENDIF

  WRITE LUN_CRT ('[MOV] Invalid queue type', NL)

  RETURN(RESULT_UNAVAILABLE)
END execute_fly_queue


ROUTINE hello : INTEGER
BEGIN
  WRITE LUN_CRT ('[MOV] Hello', NL)
  WRITE LUN_TP ('[MOV] Hello', NL)

  RETURN(RESULT_OK)
END hello


ROUTINE call_service(command : STRING) : INTEGER
VAR
  param       : ARRAY[paramlength] OF REAL
  action      : STRING[25]
  paramnumber : INTEGER
BEGIN
  action := ''
  paramnumber := 0

  get_command(command, action)

  IF action = 'Hello' THEN
    RETURN(hello)
  ENDIF

  IF action = 'clearFlyQueue' THEN
    RETURN(clear_fly_queue)
  ENDIF

  IF action = 'executeFlyQueue' THEN
    RETURN(execute_fly_queue)
  ENDIF

  paramnumber := get_parameter(command, param)

  IF paramnumber < 0 THEN
    RETURN(RESULT_PARSE_ERROR)
  ENDIF

  IF action = 'setBase' THEN
    RETURN(set_base_to(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF (action = 'setTool') OR (action = 'tool') THEN
    RETURN(set_tool_to(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF (action = 'setUframe') OR (action = 'userFrame') THEN
    RETURN(set_uframe_to(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF action = 'setSpeedLin' THEN
    RETURN(set_speed_linear_to(param[1], paramnumber))
  ENDIF

  IF action = 'setSpeedJnt' THEN
    RETURN(set_speed_joint_to(param[1], paramnumber))
  ENDIF

  IF action = 'setJointOverrides' THEN
    RETURN(set_joint_overrides(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF action = 'setAcceleration' THEN
    RETURN(set_acceleration_to(param[1], paramnumber))
  ENDIF

  IF action = 'setDeceleration' THEN
    RETURN(set_deceleration_to(param[1], paramnumber))
  ENDIF

  IF action = 'setOrientation' THEN
    RETURN(set_orientation_to(param[1], paramnumber))
  ENDIF

  IF action = 'setTermination' THEN
    RETURN(set_termination_to(param[1], paramnumber))
  ENDIF

  IF action = 'setFlyNorm' THEN
    RETURN(set_fly_norm_to(param[1], paramnumber))
  ENDIF

  IF action = 'setFlyCart' THEN
    RETURN(set_fly_cart_to(param[1], param[2], param[3], paramnumber))
  ENDIF

  IF action = 'moveJoint' THEN
    RETURN(move_joint_to(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF action = 'moveLin' THEN
    RETURN(move_linear_to(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF action = 'moveCircular' THEN
    RETURN(move_circular_to(param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], param[9], param[10], param[11], param[12], paramnumber))
  ENDIF

  IF action = 'addFlyLin' THEN
    RETURN(add_fly_linear(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  IF (action = 'addFlyCirc') OR (action = 'addFlyCircular') THEN
    RETURN(add_fly_circular(param[1], param[2], param[3], param[4], param[5], param[6], param[7], param[8], param[9], param[10], param[11], param[12], paramnumber))
  ENDIF

  IF action = 'addFlyJoint' THEN
    RETURN(add_fly_joint(param[1], param[2], param[3], param[4], param[5], param[6], paramnumber))
  ENDIF

  WRITE LUN_CRT ('[MOV] Unknown action: ', action, NL)
  WRITE LUN_TP ('[MOV] Unknown action', NL)

  RETURN(RESULT_UNKNOWN_ACTION)
END call_service


BEGIN
  parse_ok := FALSE

  fly_queue_type := QUEUE_NONE
  fly_queue_count := 0

  $FLY_TYPE := FLY_NORM
  $FLY_PER := 75

  WRITE LUN_CRT ('ROS2_MOVEMENTS mixed variable queue ready', NL)
  WRITE LUN_TP ('ROS2_MOVEMENTS mixed variable queue ready', NL)
END ROS2_MOVEMENTS
